'use strict';
/**
 * @param {number} amount
 * @param {string} currency
 * @return {number}
 */
function countCommision(amount, currency) {
 let commision = 0;
 
 if (currency == 'UAH') {
   if (amount <= 1000) {
     commision = 2;
   } else {
     commision = 1;
   }
 }
 
 if (currency == 'USD') {
   if (amount <= 5000) {
     commision = 0.5;
   } else {
     commision = 0.25;
   }
 }

 if (currency == 'EUR') {
   if (amount <= 3000) {
     commision = 0.75;
   } else {
     commision = 0;
   }
 }

 return amount * commision / 100;
}

// examples
countCommision(20000, 'USD'); // ===> 50
countCommision(500, 'UAH'); // ===> 10
countCommision(2000, 'UAH'); // ===> 20
